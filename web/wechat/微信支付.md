```typescript
// payInfo 通过请求获取

// 微信支付
minPayment: (() => {
    let flag = false
    return function() {
        // console.log("flag", flag);
        if(flag) return
        flag = true
        const { orderId } = this.orderDetail
        queryPayInfo(orderId)
        .then(res => {
            wx.requestPayment({
                ...payInfo,
                success: (res) => {
                    this.loadingtimer = setTimeout(() => {
                        uni.showLoading({
                            title: 'loading...',
                        })
                    }, 2000);
                    let count = 0
                    this.paytimer = setInterval(() => {
                        if(count >= 10) {
                            clearInterval(this.paytimer)
                            clearTimeout(this.loadingtimer)
                            uni.hideLoading()
                            flag = false
                            console.error("支付查询超时");
                            return
                        }
                        count += 1
                        // 支付结果验证接口
                        queryPayResult(orderId)
                        .then(res => {
                            if(res.status === '1') {
                                clearInterval(this.paytimer)
                                clearTimeout(this.loadingtimer)
                                uni.hideLoading()
                                console.log("支付成功");
                                flag = false
                            }
                        })
                        .catch(err => {
                            flag = false
                            console.error(err);
                        })
                    }, 2000)
                },
                fail: (res) => {
                    uni.showToast({
                        title: '支付失败',
                        icon: 'error',
                        duration: 2000,
                    })
                    flag = false
                }
            })
        })
        .catch(err => {
            console.error(err)
            flag = false
        })
    }
})(),
```

# 公众号支付

```typescript
// payInfo 通过请求获取

interface onBridgeReadyOptions <T>{
	payInfo: string;
	orderId: string;
	success?: (...args: Array<T>) => void;
	fail?: () => void;
}

/**
 * 调用微信支付
 * @param payInfo JSAPI掉漆微信支付请求参数列表
 * @param orderId 订单 ID
 * @param success 支付成功后执行
 * @param fail 支付失败后执行
 * 若 WeixinJSBridge.invoke 函数具有返回值，直接返回Promise即可 
 */
const onBridgeReady = (options: onBridgeReadyOptions<number>) => {
    // eslint-disable-next-line
    (window as any).WeixinJSBridge.invoke(
        'getBrandWCPayRequest', JSON.parse(options.payInfo),
        // eslint-disable-next-line
        function(res: any){
            if(res.err_msg == "get_brand_wcpay_request:ok" ){
                // 使用以上方式判断前端返回,微信团队郑重提示：
                //res.err_msg将在用户支付成功后返回ok，但并不保证它绝对可靠。
                loadingtimer = setTimeout(() => {
                    Toast.loading({
                        message: 'loading...',
                        forbidClick: true,
                        duration: 0,
                    })
                }, 1000)
                let count = 0
                paytimer = setInterval(() => {
                    if(count >= 10) {
                        clearInterval(paytimer)
                        clearTimeout(loadingtimer)
                        Toast.clear()
                        options.fail && options.fail()
                        console.error("支付查询超时");
                        return
                    }
                    count += 1
                    // 支付结果验证接口
                    queryPayResult(options.orderId)
                    .then(res => {
                        if(res.data.status === '1') {
                            clearInterval(paytimer)
                            clearTimeout(loadingtimer)
                            Toast.clear()
                            /**
                             * 正常情况下应该把得到的 res.data.id 返回
                             * 但是 WeixinJSBridge.invoke 是直接执行函数，并不具备返回值功能，故而只能这么调用
                             */
                            options.success && options.success(res.data.id)
                        }
                    })
                    .catch(err => {
                        console.error(err);
                    })
                }, 1000)
            } else {
                options.fail && options.fail()
                console.error("取消支付");
            }
        }
    );
}
```



# 注意问题

1. **订单状态**：微信支付的返回可能非常慢，偶尔会长达1分多钟，1分钟内用户可以取消订单。

   如果后端将微信返回后的订单直接改成已支付，则可能**将已退款的订单改成已支付**，所以修改订单状态一定要判定是不是处于待支付
